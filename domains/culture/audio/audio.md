def divisors(n):
    """Divisors. n=6 -> {1,2,3,6}"""
    return {d for d in range(1, n+1) if n % d == 0}

def sigma(n):
    """Divisor sum (OEIS A000203). sigma(6)=1+2+3+6=12"""
    return sum(divisors(n))

def tau(n):
    """Divisor count (OEIS A000005). tau(6)=4"""
    return len(divisors(n))

def phi_totient(n):
    """Euler phi (OEIS A000010). phi(6)=2"""
    return sum(1 for k in range(1, n+1) if __import__('math').gcd(k, n) == 1)

def sopfr(n):
    """Sum of prime factors (OEIS A001414). sopfr(6)=2+3=5"""
    s, k = 0, n
    for p in range(2, n+1):
        while k % p == 0:
            s += p; k //= p
        if k == 1: break
    return s

def phi_min_prime(n):
    """Smallest prime factor. phi_min(6)=2"""
    for p in range(2, n+1):

def scaling_exponent(xs, ys):

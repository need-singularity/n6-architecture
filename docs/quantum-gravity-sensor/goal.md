#!/usr/bin/env python3
"""HEXA-QGRAV n=6 양자 중력 센서 프레임워크 전수 검증"""
from math import log2, factorial, sqrt
from sympy import divisor_sigma, totient, divisor_count, factorint

def sopfr(n):
    """소인수 합 (중복 포함)"""
    return sum(p * e for p, e in factorint(n).items())

def jordan(n, k=2):
    """Jordan totient J_k"""
    result = n ** k
    for p in factorint(n):
        result = result * (1 - 1 / p ** k)
    return int(result)

def verify_qgrav(n):
    """n에 대해 양자 중력 센서 프레임워크 검증"""
    s = int(divisor_sigma(n))     # sigma
    t = int(divisor_count(n))     # tau
    p = int(totient(n))           # phi
    sp = sopfr(n)                 # sopfr
    j2 = jordan(n)                # J2

    results = {}
    # 물리 구조 검증
    results['MOT_6빔']            = (n == 6)
    results['MZ_2경로']           = (p == 2)
    results['Raman_3펄스']        = (n % p == 0) and (n // p == 3)
    results['텐서_6성분']         = (n == 6)
    results['6DOF_관성항법']      = (n == 6)
    results['4양자상태']          = (t == 4)
    results['12센서쌍']           = (s == 12)
    results['5차_전개']           = (sp == 5)
    results['24h_연속']           = (j2 == 24)
    results['60s_적분']           = (s * sp == 60)

    # 일반상대론 구조
    results['시공간_4차원']       = (t == 4)
    results['계량텐서_10성분']    = (s - p == 10)
    results['리만텐서_20성분']    = (j2 - t == 20)
    results['중력파_2편극']       = (p == 2)
    results['LISA_3위성']         = (n % p == 0) and (n // p == 3)

    # 성능 검증
    results['1296_그리드']        = (n ** t == 1296)
    results['Egyptian_에너지']    = abs(1/2 + 1/3 + 1/6 - 1.0) < 1e-10
    results['Carnot_효율']        = abs(1 - p / s - 5 / 6) < 0.001 if s > 0 else False
    results['n2=36_조화차수']     = (n ** 2 == 36)

    results['엔트로피_W']         = (s ** t == 20736)
    results['GPS_3cm']            = (n % p == 0) and (n // p == 3)

def contrast_test(n, label):
    """대조 검증"""

    exact = sum(1 for v in r.values() if v)

    print(f"{'='*55}")
    for k, v in r.items():
        mark = "EXACT" if v else "FAIL "
        print(f"  [{mark}] {k}")
    return exact, total
